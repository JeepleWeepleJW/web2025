요소 클릭 시 내용 변경

    element.innerText = '클릭됨!';
};

CSS 속성 접근 및 수정

폼 요소에 접근하기

폼 배열 사용하기

document.forms;            // 모든 폼
document.forms[0].elements; // 첫 번째 폼의 요소들

선택 목록 (<select>) 접근

const selectMenu = document.querySelector('#major');
selectMenu.options;
selectMenu.selectedIndex;
selectMenu.onchange = function() {
    alert(`[${selectMenu.options[selectMenu.selectedIndex].innerText}]를 선택했습니다.`);
};

const selectedSubject = document.querySelector("input[name='subject']:checked").value;
