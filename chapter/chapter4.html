JavaScript 함수 기초

함수

기능별로 함수를 따로 만들어두면 필요할 때마다 적절한 함수를 사용하여 실행할 수 있습니다.

function 함수명() {
    명령
}

함수명();

매개변수

외부에서 값을 받는 변수로, 함수 이름 옆 소괄호 안에 지정합니다.

function sum(a, b) {  // a, b: 매개변수
  let result = a + b;
  alert(`두 수의 합 : ${result}`);
}

인수

매개변수가 있는 함수를 호출할 때 전달하는 값을 인수(인자)라고 합니다.

sum(10, 20);  // 10, 20: 인수

return문

함수의 결과값을 반환할 때 사용합니다.

기본 매개변수

ES2015부터 매개변수에 기본값을 지정할 수 있습니다.

function multiple(a, b, c) {
  return a * b + c;
}

multiple(2, 4, 8);   // 16
multiple(2, 4);      // NaN

function multiple(a, b = 5, c = 10) {
  return a * b + c;
}

console.log(multiple(5, 10, 20));  // 70
console.log(multiple(10, 20));     // 210
console.log(multiple(10));         // 60

스코프

변수가 유효한 범위를 말합니다.

지역 스코프

함수 내부에서 선언된 변수는 함수 내에서만 사용 가능하며, 외부에서 접근하면 ReferenceError가 발생합니다.

전역 스코프

프로그램 전체에서 사용할 수 있으며, 함수 어디서나 접근 가능합니다. var 없이 선언해도 전역 변수가 됩니다.

블록 스코프

중괄호({})로 묶인 영역 내에서만 유효하며, let/const로 선언된 변수는 블록 스코프를 가집니다.

var, const, let 스코프

  • var 전역 변수는 window 객체에 저장
  • let, const 전역 변수는 스크립트 범위 내에서만 사용

익명 함수

이름 없이 변수에 할당하여 사용합니다.

let sum = function(a, b) {
  return a + b;
};

console.log(`함수 실행 결과 : ${sum(10, 20)}`);

즉시 실행 함수(IIFE)

(function(){
  // ...
})();

화살표 함수

간결한 함수 선언 방식

let hi = function() {
  return `안녕하세요?`;
};
hi();

let hi = () => { return `안녕하세요?`; };
hi();

let hi = () => `안녕하세요?`;
hi();

let hi = function(user) {
  console.log(`${user}님, 안녕하세요?`);
};
hi("홍길동");

let hi = user => console.log(`${user}님, 안녕하세요?`);
hi("홍길동");

let sum = function(a, b) {
  return a + b;
};
sum(10, 20);

let sum = (a, b) => a + b;
sum(10, 20);

콜백 함수

다른 함수의 인수로 전달하여 특정 시점에 실행되는 함수입니다.

const bttn = document.querySelector("button");  // 버튼 요소를 가져옵니다.

function display() {
  alert("클릭했습니다.");
}

bttn.addEventListener("click", display);  // 버튼을 클릭하면 display 함수를 실행합니다.

bttn.addEventListener("click", () => {    // 한 번만 사용할 때는 즉시 작성 가능
  alert("클릭했습니다.");
});

일정 시간마다 반복 setInterval()

밀리초 단위로 콜백 함수를 반복 실행합니다.

setInterval(콜백함수, 시간);

반복 실행 멈추기 clearInterval()

setInterval이 반환하는 타이머 ID를 사용해 반복을 중단합니다.

clearInterval(타이머ID);

특정 시간 이후에 실행하기 setTimeout()

지정한 시간이 지난 후 한 번만 콜백 함수를 실행합니다.

setTimeout(콜백함수, 시간);
